# wormcoast

Frontend:

  Setup:
    cd cleint/,
    pnpm install
    
  Launch: 
    pnpm start
    
  Test:
    pnpm test


Backend:

  Setup:
    cd api/,
    pnpm install
    
  Launch:
    pnpm start:dev
    
  Test:
    add your "sst.grid" file to /api/tests/resources/, 
    pnpm test:watch
